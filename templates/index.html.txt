<script>

// ✅ LIVE HASH UPDATE (while typing)
function liveUpdate(index) {

    const blocks = document.getElementsByClassName("block");

    for (let i = index; i < blocks.length; i++) {

        const data = blocks[i].querySelector("input").value;
        const prevHash = i === 0 ? "0" : document.getElementById("hash-" + (i-1)).innerText;
        const nonce = document.getElementById("nonce-" + i).innerText;

        const text = i + prevHash + data + nonce;

        const hash = CryptoJS.SHA256(text).toString();

        document.getElementById("hash-" + i).innerText = hash;
        document.getElementById("prev-" + i).innerText = prevHash;
    }
}

// ✅ MINE BUTTON
async function mine(index) {

    const res = await fetch("/mine/" + index);
    const data = await res.json();

    document.getElementById("hash-" + index).innerText = data.hash;
    document.getElementById("nonce-" + index).innerText = data.nonce;

    liveUpdate(index); // update next blocks
}
